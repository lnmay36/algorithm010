### 69. x 的平方根

	为什么能用二分查找，y=x^2(x>0),曲线单调递增
	有上下界[1,n]
	
	public int mySqrt(int x) {
        long left = 1L,right = (long)x;
        while (left <= right) {
            long mid = (left+right)/2;
            if (mid*mid==x)
                return (int)mid;
            if (mid*mid > x) {
                right = mid-1;
            }
            else 
                left = mid+1;
        }
        return (int)right;
    }

public int mySqrt(int x) {
